goals_metrics <- read_csv_if_exists(file.path(root, "data", "processed", "modeling", "goals_linear_model_metrics.csv"))
ordinal_metrics <- read_csv_if_exists(file.path(root, "data", "processed", "modeling", "result_ordinal_model_metrics.csv"))

if (nrow(goals_metrics) > 0 && nrow(ordinal_metrics) > 0) {
  data.frame(
    model = c("Goals OLS", "Ordinal result logistic"),
    target = c("team goals", "loss < draw < win"),
    rows_total = c(goals_metrics$rows_total[[1]], ordinal_metrics$rows_total[[1]]),
    primary_metric = c("test RMSE", "test accuracy"),
    value = c(
      round(goals_metrics$test_rmse[[1]], 4),
      round(ordinal_metrics$test_accuracy[[1]], 4)
    ),
    baseline_context = c(
      paste("mean test goals:", round(goals_metrics$mean_y_test[[1]], 4)),
      paste("majority baseline:", round(ordinal_metrics$baseline_majority_accuracy[[1]], 4))
    )
  )
} else {
  data.frame(note = "Model metrics are generated locally before rendering.")
}

read_csv_if_exists <- function(path) {
  if (file.exists(path)) {
    readr::read_csv(path, show_col_types = FALSE)
  } else {
    data.frame()
  }
}
