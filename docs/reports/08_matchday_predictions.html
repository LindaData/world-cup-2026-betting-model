  board$kickoff_utc <- as.POSIXct(board$kickoff_utc_iso, format = "%Y-%m-%dT%H:%M:%SZ", tz = "UTC")
  board_for_display <- board |>
    dplyr::filter(
      (
        !is.na(kickoff_utc) & kickoff_utc > Sys.time()
      ) |
        (
          is.na(kickoff_utc) & as.Date(date) >= Sys.Date() & match_timing != "Completed"
        )
    )

  display_board <- board_for_display |>
    dplyr::mutate(
      result_probability = paste0(
        "H ",
        fmt_percent(ensemble_home_win_prob, 1),
        " / D ",
        fmt_percent(ensemble_draw_prob, 1),
        " / A ",
        fmt_percent(ensemble_away_win_prob, 1)
      ),
      expected_goals = paste0(
        fmt_number(pred_home_goals_poisson, 2),
        " - ",
        fmt_number(pred_away_goals_poisson, 2)
      ),
      totals = paste0(
        "O2.5 ",
        fmt_percent(over_2_5_prob, 1),
        " / BTTS ",
        fmt_percent(both_teams_to_score_prob, 1)
      ),
      yellow_cards = dplyr::case_when(
        is.finite(home_projected_yellow_cards) & is.finite(away_projected_yellow_cards) ~
          paste0(
            fmt_number(home_projected_yellow_cards, 1),
            " - ",
            fmt_number(away_projected_yellow_cards, 1)
          ),
        TRUE ~ "Feed pending"
      ),
      lineup = paste0(home_lineup_status, " / ", away_lineup_status),
      confidence = paste0(confidence_band, " (", fmt_percent(prediction_confidence, 1), ")"),
      venue = paste(city, country, sep = ", ")
    ) |>
    dplyr::select(
      date,
      match_timing,
      match_label,
      predicted_winner,
      confidence,

The board uses three active layers:

  • Winner and outcome: an ensemble of the ordinal logistic result model and probabilities derived from the Poisson score grid.
  • Goals and likely score: the Poisson goals model produces expected goals; the score grid turns those expected goals into scoreline probabilities.
  • Totals: over 2.5 goals and both-teams-to-score probabilities come from the same Poisson score grid.

Cards and lineups are live-enrichment layers. They update when the football API returns player card history, event rows, or confirmed lineups for a fixture.

6 How To Read It

  • Predicted winner: highest ensemble probability. A draw remains possible for group-stage style matches.
  • Confidence: the model’s top probability bucket, not a guarantee.
  • Expected goals: model average score, not the most likely exact score.
  • Most likely score: the single scoreline with the highest probability in the Poisson grid.
  • O2.5: probability that total goals are 3 or more.
  • BTTS: probability that both teams score at least one goal.
